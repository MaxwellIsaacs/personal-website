// Recursive determinant calculation using cofactor expansion
public static double det(Matrix x) {
if (x.getCols() == 2) return determinant2d(x);
double determinant = 0.0;
int sign = 1;
for (int i = 0; i < x.getCols(); i++) {
// Create submatrix by removing row 0 and column i
Matrix subMatrix = createSubMatrix(x, 0, i);
determinant += sign * x.getElement(0, i) * det(subMatrix);
sign *= -1; // Alternate signs
}
return determinant;
}

// Network initialization with random weights
public NeuralNetwork(int[] layerSizes, double learningRate) {
this.layerSizes = layerSizes;
this.weights = new Matrix[layerSizes.length - 1];
this.biases = new Vector[layerSizes.length - 1];
for (int i = 0; i < weights.length; i++) {
weights[i] = Matrix.randomMatrix(
layerSizes[i+1], layerSizes[i], -1, 1);
biases[i] = Vector.randomVector(layerSizes[i], -1, 1);
}
this.learningRate = learningRate;
}
// Forward pass through all layers
public Matrix feedForward(Matrix input) {
Matrix activations = input;
for (int i = 0; i < weights.length; i++) {
activations = weights[i].multiply(activations).add(biases[i]);
this.zValues[i] = activations;
activations = activations.applyFunction(ActivationFunctions.ReLU);
}
return activations;
}
